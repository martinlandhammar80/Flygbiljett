pipeline {
    agent any

    environment {
        // Define environment variables if needed
        GITHUB_REPO_URL = 'https://github.com/your-repo/your-project.git'
        TEST_RESULTS_DIR = 'TestResults'
    }

    stages {
        stage('Clone Repository') {
            steps {
                // Checkout the repository from GitHub
                git url: "${env.GITHUB_REPO_URL}", branch: 'main'
            }
        }
        
        stage('Run NUnit Tests') {
            steps {
                script {
                    // Create a directory for test results
                    sh 'mkdir -p ${TEST_RESULTS_DIR}'

                    // Run the NUnit tests
                    def testProcess = [
                        "nohup",
                        "nunit3-console",
                        "path/to/your/test/assembly.dll",
                        "--result=${TEST_RESULTS_DIR}/TestResult.xml"
                    ].join(' ')
                    
                    // Start the NUnit test process in the background
                    sh "${testProcess} &"
                }
            }
        }
    }

    post {
        always {
            // Archive the test results
            archiveArtifacts artifacts: "${TEST_RESULTS_DIR}/**", allowEmptyArchive: true
            
            // Publish the test results
            junit "${TEST_RESULTS_DIR}/TestResult.xml"
        }
    }
}
